def predict_poly(x, koeff):
    res = 0
    #      x   
    xx = [x ** (len(koeff) - n - 1) for n in range(len(koeff))]

import numpy as np
import matplotlib.pyplot as plt

#   
def loss(w, x, y):
    M = np.dot(w, x) * y
    return 2 / (1 + np.exp(M))

#        w
def df(w, x, y):
    L1 = 1.0  #  L1-
    M = np.dot(w, x) * y
    return -2 * (1 + np.exp(M)) ** (-2) * np.exp(M) * x * y + L1 * np.sign(w)

#   (  )
x_train_14 = np.array([[16, 49],
                      [45, 28],
                      [11, 26],
                      [49, 27],
                      [44, 7],
                      [9, 33],
                      [19, 5],
                      [9, 49],
                      [6, 22],
                      [39, 32]])
